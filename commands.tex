\renewcommand{\headrulewidth}{1.4pt} % Rule under the header
\renewcommand{\footrulewidth}{1.4pt} % Rule under the footer

% Defines the command to print each word on the page, 
% \markboth{}{} prints the first word on the page 
% in the top left header and the last word in the top right
\newcommand{\entry}[2]{
	\markboth{#1}{#1}
	\textbf{#1}
	\label{#1} $\bullet$\ 
	{#2}
} 

% A helper command to add images with their sources
% first argument is the source address
% second argument is the name of the source address
% third argument is the file name
\newcommand{\fig}[3]{
	\begin{figure} [H]
		\centering
		\includegraphics[width=\linewidth]{#3}
		\caption[#1]{From~#2\protect\footnotemark}
	\end{figure}
	\footnotetext{#1}
}

\renewcommand{\ref}[1]{(\textit{See~#1})}

\newcommand{\contributor}[4]{
	\textbf{#1}\\
	\textit{#2}\\
	\small{github:~#3}\\
	\small{twitter:~#4}\\
}
